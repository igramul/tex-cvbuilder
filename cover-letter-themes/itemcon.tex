        \node [rectangle, anchor=north, fill=sidebarbackground, inner ysep=10pt, inner xsep=0pt, minimum height=1.5cm] (contact) at ($(current page.north)-(0cm,6cm)$) {%
		\begin{minipage}{.5\textwidth}
		    \vspace{2pt}
		    \hspace{2pt}{\sname\name}

		    \vspace{2mm}
		    
		    \hspace{2pt}{\sjobtitle\jobtitle}
		\end{minipage}
		\begin{minipage}{.5\textwidth}
		    \null\strut\hfill \ifthenelse{\equal{\location}{}}{}{\location\hspace{1mm}\locicon\hspace{2pt}} \\
		    \null\hfill \ifthenelse{\equal{\phone}{}}{}{\phone\hspace{1mm}\phoneicon\hspace{2pt}} \\
		    \null\hfill \ifthenelse{\equal{\mail}{}}{}{\href{mailto:\mail}{\mail}\hspace{1mm}\mailicon}\hspace{2pt}\vspace{1pt}
		\end{minipage}
        };
        \draw[line width=1pt, draw=linecolor] ($(contact.north west)$) -- ($(contact.north east)$);
        \draw[line width=1pt, draw=linecolor] ($(contact.south west)$) -- ($(contact.south east)$);
    \end{tikzpicture}
    \vspace{2\bigskipamount}
}
